# Demo

  <select-field :items="[
      { id: 1, name: 'Item A' },
      { id: 2, name: 'Item B' },
      { id: 3, name: 'Parent item',
        children: [
          { id: 4, name: 'Item C' }
        ]
      }
    ]"
    value="1"
    :empty-option="true"
  />
