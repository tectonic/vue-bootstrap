# Demo

<multicheckbox
  name="stats"
  :options="players"
  :selected-options="[1, 2]"
  :subsets="[
    { id: 'steals', name: 'Steals' },
    { id: 'rebounds', name: 'Rebounds' },
    { id: 'points', name: 'Points' }
  ]"
  :selected-subset-options="{
    steals: [],
    rebounds: [1],
    points: [1, 2]
  }"
/>
